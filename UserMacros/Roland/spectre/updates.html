===========EDIT OUT THE FOLLOWING INSTRUCTIONS FROM updates2.html===========
(create a list of clinical macros by using this command in the TOP directory)
ls -lt ./clinmacros/*.sas | gawk '{print substr($0,33)}' > clinlist.txt

(create a list of utility macros by using this command in the TOP directory)
ls -lt ./utilmacros/*.sas | gawk '{print substr($0,33)}' > utillist.txt

(create a list of system macros by using this command in the TOP directory)
ls -lt ./sysmacros/*.sas | gawk '{print substr($0,33)}' > syslist.txt

(create a list of scripts by using this command in the SCRIPT directory)
ls -lt * | fgrep -v '.' | fgrep -v '~' | gawk '{print substr($0,34)}' > scrlist.txt

(create a list of html pages by using this command in the TOP directory)
ls -lt *.html | gawk '{print substr($0,34)}' > htmlist.txt

(include these lists and convert updates.html to updates2.html by using this command)
sed -e 's%\([0-9][0-9] \)\(\./[a-z]*/\)\([a-z][a-z,0-9,_]*\.sas\)%\1<a href="\2\3" target="_blank">\3</a>%' \
-e 's%\([0-9][0-9] \)\([a-z][a-z,0-9,_]*\.html\)%\1<a href="\2" target="_blank">\2</a>%' \
-e 's%\([0-9][0-9] \)\([a-z][a-z,0-9,_]*\)\(</tt>\)%\1<a href="scripts/\2" target="_blank">\2</a>\3%' \
updates.html > updates2.html
