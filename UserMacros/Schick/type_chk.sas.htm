%macro chk_type(name);
  %let k=%scan(&name,1,-1234567890);
  %if &k=. %then %put numerics;
           %else %put character;
%mend;

%chk_type(name);
%chk_type(var23);
%chk_type(5.6);
%chk_type(1);     *finds the type of character;
%chk_type(1.0);   *type-in as a real value!;
%chk_type(-.25);
