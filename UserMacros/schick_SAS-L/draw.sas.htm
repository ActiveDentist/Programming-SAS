%MACRO  draw( x1, y1, colin, lintyp, width );

%* DRAW a line to the requested ( X1,Y1 ) coordinate.                 ;
%*--------------------------------------------------------------------;

  Y = &y1;
  LINE     = &lintyp;
  SIZE     = &width;
      IF "&colin" =: '*'  THEN ;  ELSE color = "&colin" ;
