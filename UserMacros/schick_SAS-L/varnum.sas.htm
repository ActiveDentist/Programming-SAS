%macro varnum(dsn);

  proc contents data=&dsn noprint out=_out_(keep=varnum);
  run;

  data _null_;
    set _out_ end=eof;
    if eof then call symput('varnum',trim(varnum));
  run;

%put Number of variables in data &dsn = &varnum.;

%mend varnum;
